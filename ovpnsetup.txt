function h1(text) {
  return new Paragraph({
    heading: HeadingLevel.HEADING_1,
    spacing: { before: 400, after: 200 },
    children: [new TextRun({ text, bold: true, size: 32, font: 'Arial', color: '2E4057' })]
  });
}

function h2(text) {
  return new Paragraph({
    heading: HeadingLevel.HEADING_2,
    spacing: { before: 300, after: 160 },
    children: [new TextRun({ text, bold: true, size: 26, font: 'Arial', color: '2E4057' })]
  });
}

function h3(text) {
  return new Paragraph({
    spacing: { before: 200, after: 120 },
    children: [new TextRun({ text, bold: true, size: 22, font: 'Arial', color: '3A6186' })]
  });
}

function para(text, options = {}) {
  return new Paragraph({
    spacing: { before: 80, after: 80 },
    children: [new TextRun({ text, size: 20, font: 'Arial', ...options })]
  });
}

function note(text) {
  return new Paragraph({
    spacing: { before: 100, after: 100 },
    indent: { left: 360 },
    children: [
      new TextRun({ text: '⚠️  Note: ', bold: true, size: 20, font: 'Arial', color: 'C0392B' }),
      new TextRun({ text, size: 20, font: 'Arial', italics: true })
    ]
  });
}

function tip(text) {
  return new Paragraph({
    spacing: { before: 100, after: 100 },
    indent: { left: 360 },
    children: [
      new TextRun({ text: '✅  ', size: 20, font: 'Arial' }),
      new TextRun({ text, size: 20, font: 'Arial', italics: true, color: '27AE60' })
    ]
  });
}

function bullet(text, level = 0) {
  return new Paragraph({
    numbering: { reference: 'bullets', level },
    spacing: { before: 60, after: 60 },
    children: [new TextRun({ text, size: 20, font: 'Arial' })]
  });
}

function code(text) {
  return new Paragraph({
    spacing: { before: 60, after: 60 },
    indent: { left: 360 },
    border: { left: { style: BorderStyle.SINGLE, size: 6, color: '3A6186' } },
    children: [new TextRun({ text, size: 18, font: 'Courier New', color: '2C3E50' })]
  });
}

function codeBlock(lines) {
  return lines.map(l => code(l));
}

function tableRow(cells, isHeader = false) {
  return new TableRow({
    tableHeader: isHeader,
    children: cells.map(({ text, width }) => new TableCell({
      borders: isHeader ? headerBorders : borders,
      width: { size: width, type: WidthType.DXA },
      shading: isHeader
        ? { fill: '2E4057', type: ShadingType.CLEAR }
        : { fill: 'F8F9FA', type: ShadingType.CLEAR },
      margins: { top: 80, bottom: 80, left: 120, right: 120 },
      children: [new Paragraph({
        children: [new TextRun({
          text,
          size: 18,
          font: 'Arial',
          bold: isHeader,
          color: isHeader ? 'FFFFFF' : '2C3E50'
        })]
      })]
    }))
  });
}

function spacer() {
  return new Paragraph({ spacing: { before: 120, after: 120 }, children: [new TextRun('')] });
}

function sectionDivider() {
  return new Paragraph({
    spacing: { before: 200, after: 200 },
    border: { bottom: { style: BorderStyle.SINGLE, size: 4, color: '2E4057' } },
    children: [new TextRun('')]
  });
}
